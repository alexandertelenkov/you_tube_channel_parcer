# Пример файла с прокси
# Формат: IP:PORT или protocol://IP:PORT
# Поддерживаются: HTTP, HTTPS, SOCKS4, SOCKS5

# HTTP прокси
proxy1.example.com:8080
192.168.1.100:3128

# HTTPS прокси
https://proxy2.example.com:8443

# SOCKS5 прокси (рекомендуется для YouTube)
socks5://proxy3.example.com:1080
socks5://45.123.45.67:1080

# Если протокол не указан, используется HTTP
23.45.67.89:8888
34.56.78.90:9999

# Можно указывать порты 9Proxy
127.0.0.1:60000
127.0.0.1:60001
127.0.0.1:60002
127.0.0.1:60003
127.0.0.1:60004
